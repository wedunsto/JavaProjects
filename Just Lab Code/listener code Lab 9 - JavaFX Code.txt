//Returns the Maze character associated with the given button   
private char getColorChar(Button b )
    {
         if (b.getStyle().indexOf("blue") > -1) {
                    return '0';
                }
                else if (b.getStyle().indexOf("white") > -1) {
                    return '1';
                }
                else if (b.getStyle().indexOf("green") > -1) {
                    return 'S';
                }
                else {
                    return 'E';
                }
    }
